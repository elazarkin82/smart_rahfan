# =====================================================================
# Execution Examples for the Fully Convolutional Context-Preserving Tracker (TargetTracker)
# =====================================================================

# ---------------------------------------------------------------------
# 1. Video-Based Siamese Training Dataset Generation
# ---------------------------------------------------------------------

# A. Run in interactive real-time HUD visualization/preview mode:
#    (Renders SIFT matches and tracking targets in real-time, no files saved.
#     Controls: Press [SPACE] for next frame, [ESC] or [Q] to exit.)
python3 training/tracker/tracker_ver3/dataset_generation_project/dataset_generator_from_video.py /path/to/your/videos -v

# B. Run in dataset generation mode (saves serialized 2-channel pickle batches):
python3 training/tracker/tracker_ver3/dataset_generation_project/dataset_generator_from_video.py /path/to/your/videos -o video_dataset --num_of_samples 16384 --batch_size 256

# C. Alternatively, run the centralized shell pipeline loading configurations from pipeline_config.json:
bash training/tracker/tracker_ver3/dataset_generation_project/run_generate.sh


# ---------------------------------------------------------------------
# 2. Target Tracker FCN Model Training
# ---------------------------------------------------------------------

# A. Train directly via CLI utilizing Dice + Binary Cross Entropy (dice_bce) Loss:
python3 training/tracker/tracker_ver3/tracker_model.py train \
    --dataset_dir video_dataset \
    --lr 0.001 \
    --num_of_epochs 100 \
    --loss dice_bce \
    --eval_pkl_num 4 \
    --output outputs/tracker.keras \
    --best_train_loss_output outputs/tracker_best_train_loss.keras \
    --log_file train.log

# B. Alternatively, execute training using the pre-configured training shell script:
bash training/tracker/tracker_ver3/run_tracker_training.sh


# ---------------------------------------------------------------------
# 3. Keras FCN Heatmap Visual Inspector (GUI)
# ---------------------------------------------------------------------

# Run Tkinter GUI to inspect 2-channel pickles, run live inference,
# and render expected/predicted spatial probability heatmaps side by side:
python3 training/tracker/tracker_ver3/keras_visualization_test.py \
    --dataset_dir video_dataset/ \
    --model_path outputs/tracker.keras
